    <div class="container">
        <h2>Canvas 工具槽绘制</h2>
        <canvas id="canvas"></canvas>
        
        <div class="controls">
            <button onclick="toggleSlot(0)">切换槽 0 选中</button>
            <button onclick="toggleSlot(1)">切换槽 1 禁用</button>
            <button onclick="toggleSlot(2)">切换槽 2 选中</button>
            <button onclick="redraw()">重绘</button>
        </div>

        <div class="info">
            <strong>说明：</strong><br>
            • 长按点击画布可检测槽位（demo 用）<br>
            • 支持自定义槽位数、大小、颜色、图标<br>
            • 可扩展为：禁用状态、数量提示、拖拽等功能
        </div>

    <script>
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');

        // ========== 配置 ==========
        const config = {
            canvasWidth: 600,
            canvasHeight: 150,
            slotCount: 5,
            slotSize: 80,
            slotGap: 15,
            padding: 20,
            // 状态追踪
            selectedSlots: new Set(),
            disabledSlots: new Set(),
        };

        // 物品数据（演示）
        const items = [
            { name: '金块', color: '#ffd700', icon: 'drawGoldBar' },
            { name: '扫帚', color: '#d4882d', icon: 'drawBrush' },
            { name: '喷雾', color: '#5b9bff', icon: 'drawSpray' },
            { name: '炸弹', color: '#6b7b9f', icon: 'drawBomb' },
            { name: '箱子', color: '#c9a95c', icon: 'drawBox' },
        ];

        // ========== 初始化 Canvas ==========
        function initCanvas() {
            canvas.width = config.canvasWidth;
            canvas.height = config.canvasHeight;
            canvas.style.cursor = 'pointer';
            redraw();
            canvas.addEventListener('click', handleCanvasClick);
        }

        // ========== 主绘制函数 ==========
        function redraw() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            drawContainerBackground();
            drawSlots();
        }

        // ========== 1) 容器背景 ==========
        function drawContainerBackground() {
            const { canvasWidth, canvasHeight, slotCount, slotSize, slotGap, padding } = config;

            // 计算容器宽度
            const innerW = slotCount * slotSize + (slotCount - 1) * slotGap;
            const containerW = innerW + padding * 2;
            const containerH = slotSize + padding * 2;
            const containerX = (canvasWidth - containerW) / 2;
            const containerY = (canvasHeight - containerH) / 2;

            // 圆角矩形背景
            const radius = 15;
            drawRoundRect(ctx, containerX, containerY, containerW, containerH, radius);
            
            // 主背景色（棕色/金色）
            const bgGrad = ctx.createLinearGradient(containerX, containerY, containerX, containerY + containerH);
            bgGrad.addColorStop(0, '#c9905f');
            bgGrad.addColorStop(0.5, '#a86f38');
            bgGrad.addColorStop(1, '#8b5a2b');
            ctx.fillStyle = bgGrad;
            ctx.fill();

            // 外边框（深色）
            ctx.strokeStyle = 'rgba(60, 30, 10, 0.6)';
            ctx.lineWidth = 3;
            ctx.stroke();

            // 内高光（可选）
            ctx.save();
            ctx.globalAlpha = 0.25;
            const highlightGrad = ctx.createLinearGradient(containerX, containerY, containerX, containerY + containerH * 0.3);
            highlightGrad.addColorStop(0, 'rgba(255, 255, 255, 0.8)');
            highlightGrad.addColorStop(1, 'rgba(255, 255, 255, 0)');
            drawRoundRect(ctx, containerX + 2, containerY + 2, containerW - 4, containerH * 0.25, radius - 1);
            ctx.fillStyle = highlightGrad;
            ctx.fill();
            ctx.restore();

            // 存储容器位置（供槽位使用）
            config.containerX = containerX;
            config.containerY = containerY;
            config.containerW = containerW;
            config.containerH = containerH;
        }

        // ========== 2) 绘制所有槽位 ==========
        function drawSlots() {
            const { canvasWidth, canvasHeight, slotCount, slotSize, slotGap, padding } = config;
            const { containerX, containerY } = config;

            const innerW = slotCount * slotSize + (slotCount - 1) * slotGap;
            const startX = containerX + padding + (config.containerW - padding * 2 - innerW) / 2;
            const startY = containerY + padding;

            for (let i = 0; i < slotCount; i++) {
                const x = startX + i * (slotSize + slotGap);
                const y = startY;

                drawSlot(x, y, slotSize, i);
            }
        }

        // ========== 3) 绘制单个槽位 ==========
        function drawSlot(x, y, size, index) {
            const isSelected = config.selectedSlots.has(index);
            const isDisabled = config.disabledSlots.has(index);

            // 槽位背景
            const slotGrad = ctx.createLinearGradient(x, y, x, y + size);
            slotGrad.addColorStop(0, isDisabled ? '#7a7a7a' : '#e8d4b8');
            slotGrad.addColorStop(0.5, isDisabled ? '#5a5a5a' : '#d9c399');
            slotGrad.addColorStop(1, isDisabled ? '#3a3a3a' : '#b8956b');

            ctx.save();

            // 选中状态：添加发光边框
            if (isSelected) {
                ctx.shadowColor = 'rgba(255, 215, 0, 0.8)';
                ctx.shadowBlur = 15;
                ctx.shadowOffsetX = 0;
                ctx.shadowOffsetY = 0;
            }

            drawRoundRect(ctx, x, y, size, size, 8);
            ctx.fillStyle = slotGrad;
            ctx.fill();

            // 槽位边框
            const borderColor = isSelected ? '#ffd700' : isDisabled ? '#555' : '#8b7355';
            ctx.strokeStyle = borderColor;
            ctx.lineWidth = 2;
            ctx.stroke();

            // 槽位内阴影
            ctx.shadowColor = 'rgba(0, 0, 0, 0.3)';
            ctx.shadowBlur = 3;
            ctx.shadowInset = true;
            drawRoundRect(ctx, x + 2, y + 2, size - 4, size - 4, 6);
            ctx.strokeStyle = 'rgba(0, 0, 0, 0.15)';
            ctx.lineWidth = 1;
            ctx.stroke();

            ctx.restore();

            // 禁用时添加斜纹
            if (isDisabled) {
                drawDisabledPattern(x, y, size);
            }

            // 绘制物品图标
            const item = items[index];
            if (item) {
                drawItemIcon(x, y, size, item, isDisabled);
            }

            // 存储槽位位置（供点击检测）
            if (!config.slotPositions) config.slotPositions = [];
            config.slotPositions[index] = { x, y, size };
        }

        // ========== 4) 禁用状态斜纹 ==========
        function drawDisabledPattern(x, y, size) {
            ctx.save();
            ctx.globalAlpha = 0.2;
            ctx.strokeStyle = '#000';
            ctx.lineWidth = 2;

            for (let i = -size; i < size * 2; i += 8) {
                ctx.beginPath();
                ctx.moveTo(x + i, y);
                ctx.lineTo(x + i + size, y + size);
                ctx.stroke();
            }

            ctx.restore();
        }

        // ========== 5) 绘制物品图标 ==========
        function drawItemIcon(x, y, size, item, isDisabled) {
            const iconX = x + size / 2;
            const iconY = y + size / 2;
            const iconSize = size * 0.55;

            ctx.save();
            if (isDisabled) ctx.globalAlpha = 0.4;

            // 根据物品类型调用对应绘制函数
            if (typeof window[item.icon] === 'function') {
                window[item.icon](iconX, iconY, iconSize, item.color);
            }

            ctx.restore();
        }

        // ========== 图标绘制函数（可扩展）==========
        function drawGoldBar(cx, cy, size, color) {
            ctx.save();
            ctx.fillStyle = color;
            ctx.beginPath();
            ctx.rect(cx - size * 0.4, cy - size * 0.5, size * 0.8, size);
            ctx.fill();

            // 高光
            ctx.fillStyle = 'rgba(255, 255, 255, 0.4)';
            ctx.beginPath();
            ctx.rect(cx - size * 0.35, cy - size * 0.45, size * 0.2, size * 0.4);
            ctx.fill();

            ctx.restore();
        }

        function drawBrush(cx, cy, size, color) {
            ctx.save();
            // 刷毛
            ctx.fillStyle = color;
            ctx.beginPath();
            ctx.ellipse(cx, cy - size * 0.2, size * 0.35, size * 0.3, 0, 0, Math.PI * 2);
            ctx.fill();

            // 手柄
            ctx.fillStyle = '#8b5a2b';
            ctx.beginPath();
            ctx.rect(cx - size * 0.15, cy - size * 0.15, size * 0.3, size * 0.5);
            ctx.fill();

            ctx.restore();
        }

        function drawSpray(cx, cy, size, color) {
            ctx.save();
            // 瓶身
            ctx.fillStyle = color;
            ctx.beginPath();
            ctx.rect(cx - size * 0.25, cy - size * 0.15, size * 0.5, size * 0.5);
            ctx.fill();

            // 喷嘴
            ctx.fillStyle = '#333';
            ctx.beginPath();
            ctx.rect(cx - size * 0.2, cy - size * 0.35, size * 0.4, size * 0.25);
            ctx.fill();

            // 高光
            ctx.fillStyle = 'rgba(255, 255, 255, 0.5)';
            ctx.beginPath();
            ctx.rect(cx - size * 0.2, cy - size * 0.1, size * 0.15, size * 0.3);
            ctx.fill();

            ctx.restore();
        }

        function drawBomb(cx, cy, size, color) {
            ctx.save();
            // 球体
            ctx.fillStyle = color;
            ctx.beginPath();
            ctx.arc(cx, cy, size * 0.35, 0, Math.PI * 2);
            ctx.fill();

            // 引信
            ctx.strokeStyle = '#d4a574';
            ctx.lineWidth = 3;
            ctx.beginPath();
            ctx.moveTo(cx, cy - size * 0.35);
            ctx.quadraticCurveTo(cx - size * 0.15, cy - size * 0.55, cx - size * 0.05, cy - size * 0.65);
            ctx.stroke();

            ctx.restore();
        }

        function drawBox(cx, cy, size, color) {
            ctx.save();
            ctx.fillStyle = color;

            // 前面
            ctx.beginPath();
            ctx.rect(cx - size * 0.35, cy - size * 0.2, size * 0.7, size * 0.5);
            ctx.fill();

            // 顶部
            ctx.fillStyle = '#daa02a';
            ctx.beginPath();
            ctx.moveTo(cx - size * 0.35, cy - size * 0.2);
            ctx.lineTo(cx - size * 0.25, cy - size * 0.35);
            ctx.lineTo(cx + size * 0.45, cy - size * 0.35);
            ctx.lineTo(cx + size * 0.35, cy - size * 0.2);
            ctx.closePath();
            ctx.fill();

            // 边框
            ctx.strokeStyle = 'rgba(0, 0, 0, 0.3)';
            ctx.lineWidth = 2;
            ctx.stroke();

            ctx.restore();
        }

        // ========== 工具函数 ==========
        function drawRoundRect(ctx, x, y, w, h, r) {
            const radius = Math.min(r, w / 2, h / 2);
            ctx.beginPath();
            ctx.moveTo(x + radius, y);
            ctx.arcTo(x + w, y, x + w, y + h, radius);
            ctx.arcTo(x + w, y + h, x, y + h, radius);
            ctx.arcTo(x, y + h, x, y, radius);
            ctx.arcTo(x, y, x + w, y, radius);
            ctx.closePath();
        }

        // ========== 交互 ==========
        function handleCanvasClick(e) {
            const rect = canvas.getBoundingClientRect();
            const clickX = e.clientX - rect.left;
            const clickY = e.clientY - rect.top;

            // 检测点击的槽位
            for (let i = 0; i < config.slotCount; i++) {
                const slot = config.slotPositions[i];
                if (clickX >= slot.x && clickX <= slot.x + slot.size &&
                    clickY >= slot.y && clickY <= slot.y + slot.size) {
                    // 点击相同槽位不取消，点击不同槽位则切换
                    if (!config.selectedSlots.has(i)) {
                        config.selectedSlots.clear();
                        config.selectedSlots.add(i);
                        console.log(`选中槽位: ${i} (${items[i].name})`);
                        redraw();
                    }
                    break;
                }
            }
        }

        function toggleSlot(index) {
            // 单选模式：清空所有选中，然后选中这个
            config.selectedSlots.clear();
            config.selectedSlots.add(index);
            console.log(`选中槽位: ${index} (${items[index].name})`);
            redraw();
        }

        // ========== 初始化 ==========
        initCanvas();
    </script>
