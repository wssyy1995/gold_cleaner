<body>
  <div class="game-scene">
    <h1 class="game-title">✨ 清洁大师游戏</h1>
    <p class="game-desc">点击下方按钮，体验获得新工具的惊喜时刻！这个弹窗完全使用 Canvas + JavaScript 实现，包含动画星星效果和交互式按钮。</p>

  <script>
    class StarParticle {
      constructor(x, y) {
        this.x = x;
        this.y = y;
        this.size = Math.random() * 3 + 1;
        this.speedX = (Math.random() - 0.5) * 4;
        this.speedY = (Math.random() - 0.5) * 2 - 2; // 向上飘
        this.alpha = 1;
        this.decay = Math.random() * 0.02 + 0.01;
        this.color = `hsl(${Math.random() * 60 + 40}, 100%, ${Math.random() * 30 + 70}%)`;
      }
      
      update() {
        this.x += this.speedX;
        this.y += this.speedY;
        this.alpha -= this.decay;
        this.size *= 0.98;
      }
      
      draw(ctx) {
        ctx.save();
        ctx.globalAlpha = this.alpha;
        ctx.fillStyle = this.color;
        
        // 绘制星星形状
        ctx.beginPath();
        for (let i = 0; i < 5; i++) {
          const angle = (i * Math.PI * 2) / 5 - Math.PI / 2;
          const radius = this.size;
          const x = this.x + Math.cos(angle) * radius;
          const y = this.y + Math.sin(angle) * radius;
          if (i === 0) ctx.moveTo(x, y);
          else ctx.lineTo(x, y);
          
          const innerAngle = angle + Math.PI / 5;
          const innerX = this.x + Math.cos(innerAngle) * (radius * 0.4);
          const innerY = this.y + Math.sin(innerAngle) * (radius * 0.4);
          ctx.lineTo(innerX, innerY);
        }
        ctx.closePath();
        ctx.fill();
        
        ctx.restore();
      }
      
      isOffScreen(canvasHeight) {
        return this.alpha <= 0 || this.y < -50;
      }
    }

    class Popup {
      constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.width = 0;
        this.height = 0;
        this.stars = [];
        this.lastTime = 0;
        this.buttonHover = false;
        this.buttonActive = false;
        this.animationFrame = null;
        this.scale = 0;
        this.init();
      }
      
      init() {
        this.resize();
        window.addEventListener('resize', () => this.resize());
        this.canvas.addEventListener('mousemove', (e) => this.handleMouseMove(e));
        this.canvas.addEventListener('click', (e) => this.handleClick(e));
        this.canvas.addEventListener('mousedown', () => this.buttonActive = true);
        this.canvas.addEventListener('mouseup', () => this.buttonActive = false);
        this.canvas.addEventListener('mouseout', () => {
          this.buttonHover = false;
          this.buttonActive = false;
        });
        
        // 初始星星
        for (let i = 0; i < 30; i++) {
          this.createStar();
        }
      }
      
      resize() {
        const maxWidth = Math.min(500, window.innerWidth * 0.9);
        const maxHeight = Math.min(400, window.innerHeight * 0.8);
        
        this.width = maxWidth;
        this.height = maxHeight;
        this.canvas.width = maxWidth;
        this.canvas.height = maxHeight;
        
        // 重新定位弹窗
        this.canvas.style.width = `${maxWidth}px`;
        this.canvas.style.height = `${maxHeight}px`;
      }
      
      createStar() {
        const x = Math.random() * this.width;
        const y = this.height * 0.3 + Math.random() * this.height * 0.2;
        this.stars.push(new StarParticle(x, y));
      }
      
      handleMouseMove(e) {
        const rect = this.canvas.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        // 检测是否在按钮上
        const buttonY = this.height * 0.85;
        const buttonWidth = this.width * 0.8;
        const buttonHeight = this.height * 0.12;
        const buttonX = (this.width - buttonWidth) / 2;
        
        this.buttonHover = (
          x >= buttonX && 
          x <= buttonX + buttonWidth && 
          y >= buttonY && 
          y <= buttonY + buttonHeight
        );
      }
      
      handleClick(e) {
        const rect = this.canvas.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        // 检测按钮点击
        const buttonY = this.height * 0.85;
        const buttonWidth = this.width * 0.8;
        const buttonHeight = this.height * 0.12;
        const buttonX = (this.width - buttonWidth) / 2;
        
        if (this.buttonHover && x >= buttonX && x <= buttonX + buttonWidth && y >= buttonY && y <= buttonY + buttonHeight) {
          this.closePopup();
        }
      }
      
      closePopup() {
        const popupContainer = document.getElementById('popupContainer');
        popupContainer.classList.remove('show');
        
        // 2秒后重置
        setTimeout(() => {
          if (this.animationFrame) {
            cancelAnimationFrame(this.animationFrame);
          }
          this.stars = [];
          for (let i = 0; i < 30; i++) {
            this.createStar();
          }
          this.scale = 0;
        }, 300);
      }
      
      update(timestamp) {
        if (!this.lastTime) this.lastTime = timestamp;
        const deltaTime = timestamp - this.lastTime;
        this.lastTime = timestamp;
        
        // 创建新星星
        if (Math.random() > 0.8) {
          this.createStar();
        }
        
        // 更新星星
        for (let i = this.stars.length - 1; i >= 0; i--) {
          this.stars[i].update();
          if (this.stars[i].isOffScreen(this.height)) {
            this.stars.splice(i, 1);
          }
        }
        
        // 动画缩放效果
        if (this.scale < 1) {
          this.scale += 0.05;
        }
        
        this.draw();
        this.animationFrame = requestAnimationFrame((time) => this.update(time));
      }
      
      draw() {
        const ctx = this.ctx;
        ctx.clearRect(0, 0, this.width, this.height);
        
        // 整体缩放动画
        ctx.save();
        ctx.translate(this.width / 2, this.height / 2);
        ctx.scale(this.scale, this.scale);
        ctx.translate(-this.width / 2, -this.height / 2);
        
        // 1. 绘制弹窗背景
        const popupPadding = this.width * 0.08;
        const popupWidth = this.width - popupPadding * 2;
        const popupHeight = this.height - popupPadding * 2;
        const popupX = popupPadding;
        const popupY = popupPadding;
        const popupRadius = 24;
        
        // 阴影
        ctx.shadowColor = 'rgba(0, 0, 0, 0.4)';
        ctx.shadowBlur = 30;
        ctx.shadowOffsetY = 10;
        
        // 弹窗背景
        ctx.fillStyle = 'white';
        this.drawRoundedRect(ctx, popupX, popupY, popupWidth, popupHeight, popupRadius);
        ctx.fill();
        
        ctx.shadowColor = 'transparent';
        ctx.shadowBlur = 0;
        
        // 2. 顶部星星装饰区域
        const headerHeight = this.height * 0.25;
        const headerY = popupY;
        
        // 顶部渐变背景
        const headerGrad = ctx.createLinearGradient(popupX, headerY, popupX, headerY + headerHeight);
        headerGrad.addColorStop(0, '#ff9a9e');
        headerGrad.addColorStop(0.5, '#fad0c4');
        headerGrad.addColorStop(1, '#fbc2eb');
        
        ctx.fillStyle = headerGrad;
        this.drawRoundedRect(ctx, popupX, headerY, popupWidth, headerHeight, popupRadius);
        ctx.fill();
        
        // 3. 标题
        ctx.fillStyle = '#8b4513';
        ctx.font = `bold ${Math.floor(this.width * 0.08)}px 'Arial', sans-serif`;
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.fillText('🎉 获得新工具！', this.width / 2, headerY + headerHeight / 2);
        
        // 4. 工具卡片
        const cardWidth = popupWidth * 0.7;
        const cardHeight = popupHeight * 0.4;
        const cardX = popupX + (popupWidth - cardWidth) / 2;
        const cardY = headerY + headerHeight + popupPadding * 0.5;
        
        // 卡片背景
        ctx.fillStyle = '#fff8dc';
        ctx.strokeStyle = '#d4af37';
        ctx.lineWidth = 3;
        this.drawRoundedRect(ctx, cardX, cardY, cardWidth, cardHeight, 16);
        ctx.fill();
        ctx.stroke();
        
        // 卡片阴影
        ctx.shadowColor = 'rgba(0, 0, 0, 0.1)';
        ctx.shadowBlur = 10;
        ctx.shadowOffsetY = 5;
        ctx.fillStyle = 'rgba(0,0,0,0.05)';
        this.drawRoundedRect(ctx, cardX + 5, cardY + 5, cardWidth - 10, cardHeight - 10, 12);
        ctx.fill();
        ctx.shadowColor = 'transparent';
        ctx.shadowBlur = 0;
        
        // 绘制工具图标（海绵）
        const toolSize = Math.min(cardWidth * 0.6, cardHeight * 0.6);
        const toolX = cardX + cardWidth / 2;
        const toolY = cardY + cardHeight / 2;
        
        // 海绵主体
        ctx.fillStyle = '#ffd700';
        ctx.beginPath();
        ctx.ellipse(toolX, toolY, toolSize * 0.6, toolSize * 0.4, 0, 0, Math.PI * 2);
        ctx.fill();
        
        // 海绵细节
        ctx.fillStyle = '#ffcc00';
        ctx.beginPath();
        ctx.ellipse(toolX - toolSize * 0.15, toolY - toolSize * 0.05, toolSize * 0.2, toolSize * 0.15, 0, 0, Math.PI * 2);
        ctx.fill();
        
        ctx.fillStyle = '#ffaa00';
        ctx.beginPath();
        ctx.ellipse(toolX + toolSize * 0.15, toolY + toolSize * 0.05, toolSize * 0.15, toolSize * 0.12, 0, 0, Math.PI * 2);
        ctx.fill();
        
        // 5. 按钮
        const buttonWidth = popupWidth * 0.8;
        const buttonHeight = this.height * 0.12;
        const buttonX = popupX + (popupWidth - buttonWidth) / 2;
        const buttonY = popupY + popupHeight - buttonHeight - popupPadding * 0.5;
        
        // 按钮状态
        const buttonColor = this.buttonActive ? '#e0b34a' : (this.buttonHover ? '#f5c542' : '#ffd700');
        const buttonShadow = this.buttonActive ? 2 : 4;
        
        // 按钮阴影
        ctx.shadowColor = 'rgba(0, 0, 0, 0.25)';
        ctx.shadowBlur = 15;
        ctx.shadowOffsetY = buttonShadow;
        
        // 按钮背景
        ctx.fillStyle = buttonColor;
        this.drawRoundedRect(ctx, buttonX, buttonY, buttonWidth, buttonHeight, 18);
        ctx.fill();
        
        ctx.shadowColor = 'transparent';
        ctx.shadowBlur = 0;
        
        // 按钮边框
        ctx.strokeStyle = '#8b4513';
        ctx.lineWidth = 3;
        this.drawRoundedRect(ctx, buttonX, buttonY, buttonWidth, buttonHeight, 18);
        ctx.stroke();
        
        // 按钮文字
        ctx.fillStyle = '#8b4513';
        ctx.font = `bold ${Math.floor(this.width * 0.06)}px 'Arial', sans-serif`;
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.fillText('✨ 收 取 ✨', buttonX + buttonWidth / 2, buttonY + buttonHeight / 2);
        
        // 6. 绘制所有星星
        for (const star of this.stars) {
          star.draw(ctx);
        }
        
        ctx.restore();
      }
      
      drawRoundedRect(ctx, x, y, width, height, radius) {
        ctx.beginPath();
        ctx.moveTo(x + radius, y);
        ctx.arcTo(x + width, y, x + width, y + height, radius);
        ctx.arcTo(x + width, y + height, x, y + height, radius);
        ctx.arcTo(x, y + height, x, y, radius);
        ctx.arcTo(x, y, x + width, y, radius);
        ctx.closePath();
      }
      
      show() {
        this.scale = 0;
        this.animationFrame = requestAnimationFrame((time) => this.update(time));
      }
    }

    // 初始化
    document.addEventListener('DOMContentLoaded', () => {
      const popupContainer = document.getElementById('popupContainer');
      const popupCanvas = document.getElementById('popupCanvas');
      const showPopupBtn = document.getElementById('showPopup');
      
      // 创建弹窗实例
      const popup = new Popup(popupCanvas);
      
      // 显示弹窗
      showPopupBtn.addEventListener('click', () => {
        popupContainer.classList.add('show');
        setTimeout(() => {
          popup.show();
        }, 100);
      });
      
      // 点击遮罩关闭
      popupContainer.addEventListener('click', (e) => {
        if (e.target === popupContainer) {
          popup.closePopup();
        }
      });
    });
  </script>
